أنواع البيانات
أربعة أنواع تكفيك الآن 📦
كل قيمة في Python لها نوع — والنوع يحدد ما يمكنك فعله بها

int — عدد صحيح

humidity = 65

أعداد بدون فاصلة — عدد الحساسات، عدد التنبيهات

float — عدد عشري

temperature = 28.5

أعداد بفاصلة — قراءات الحرارة والرطوبة الدقيقة

str — نص

status = "طبيعي"

أي نص بين علامتَي تنصيص — الرسائل والتسميات

bool — منطقي

rain = True

قيمتان فقط: True أو False — هل يمطر؟ هل هناك تنبيه؟

المتغيرات بعمق
قراءات الحساسات كمتغيرات 🌡
هكذا يُخزّن رصد (Rasd) البيانات — متغير لكل حساس
# قراءات حساسات رصد (Rasd) temperature = 28.5 # float — الحرارة بالسيلزيوس humidity = 65 # int — الرطوبة بالنسبة المئوية water_level = 62000 # int — قيمة عالية = جاف soil_moisture = 18000 # int — قيمة منخفضة = رطب rain_detected = False # bool — هل يمطر؟ location = "غرفة الصف" # str — اسم الموقع # طباعة تقرير print(f"الموقع: {location}") print(f"الحرارة: {temperature}°C | الرطوبة: {humidity}%")
💡 الـ f-string يسمح لك بدمج المتغيرات مع النص بسهولة — f"النص {المتغير}"
تحويل الأنواع
input() دائماً يُرجع نصاً ⚠️
هذا الخطأ الأكثر شيوعاً للمبتدئين — وسنحله الآن للأبد
# ❌ خطأ شائع temp = input("الحرارة؟ ") if temp > 35: # TypeError: لا يمكن مقارنة نص برقم! print("حار") # ✅ الصحيح — حوّل النص إلى رقم temp = float(input("الحرارة؟ ")) if temp > 35: # ✓ يعمل بشكل صحيح print("🔴 حار جداً") # أدوات التحويل: # int("42") → 42 | float("28.5") → 28.5 # str(28) → "28" | bool(0) → False
تطبيق ١
محاكي الحساس الكامل 🌡
اكتب هذا البرنامج على Trinket — ثم عدّله وجرّب
# محاكي حساس رصد (Rasd) temp = float(input("درجة الحرارة (°C): ")) humid = int(input("الرطوبة (%): ")) print("\n====== تقرير رصد ======") print(f"الحرارة : {temp}°C") print(f"الرطوبة : {humid}%") print("========================") if temp > 35: print("🔴 تنبيه: حرارة عالية") elif temp < 10: print("🔵 تنبيه: برودة شديدة") else: print("✅ الحرارة طبيعية") if humid > 80: print("💧 تنبيه: رطوبة مرتفعة") elif humid < 20: print("🏜 تنبيه: جفاف") else: print("✅ الرطوبة طبيعية")
تطبيق ٢
محوّل الحرارة 🌡↔️🌡
الحساسات الأمريكية ترسل فهرنهايت — رصد (Rasd) يحتاج سيلزيوس
# محوّل درجات الحرارة unit = input("أدخل الوحدة (C/F): ").upper() temp = float(input("أدخل درجة الحرارة: ")) if unit == "C": result = (temp * 9/5) + 32 print(f"{temp}°C = {result:.1f}°F") elif unit == "F": result = (temp - 32) * 5/9 print(f"{temp}°F = {result:.1f}°C") else: print("وحدة غير معروفة! استخدم C أو F")
🔍 جديد: .upper() تحوّل الحروف لكبيرة · :.1f تُقيّد الأرقام العشرية برقم واحد
تطبيق ٣ — جماعي
لوحة تقرير رصد (Rasd) النصية 📊
نكتبها معاً على السبورة — ثم كل طالب يُشغّلها على Trinket
# لوحة تقرير نصية — مثل dashboard رصد لكن في Terminal sensors = [ ("الحرارة", float(input("الحرارة°C: ")), 10, 35), ("الرطوبة", int(input("الرطوبة%: ")), 20, 80), ] print("\n╔══════════════════════╗") print("║ تقرير رصد (Rasd) ║") print("╚══════════════════════╝") for name, val, mn, mx in sensors: status = "✅ طبيعي" if mn <= val <= mx else "⚠️ تنبيه" print(f" {name}: {val} → {status}")
