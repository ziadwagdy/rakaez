الحلقات — النوع الأول
حلقة for 🔄
تكرر الكود لعدد محدد مسبقاً من المرات — مثالية لقراءات الحساسات!
# اطبع الأرقام من 1 إلى 5
for i in range(1, 6):
    print(f"العدد: {i}")

# محاكاة 5 قراءات من الحساس
import random
for i in range(5):
    temp = round(random.uniform(20, 40), 1)
    print(f"القراءة {i+1}: {temp}°C")

# range(start, stop, step) — الخطوة اختيارية
for t in range(0, 100, 10):  # 0,10,20...90
    print(f"درجة: {t}")
📝 range(stop): من 0 إلى stop-1 | range(start, stop): من start إلى stop-1
الحلقات — النوع الثاني
حلقة while ⏳
تكرر طالما الشرط صحيح — هكذا يعمل رصد (Rasd) فعلاً!
# المراقبة المستمرة — نمط رصد (Rasd) الحقيقي
import random, time

monitoring = True
count = 0

while monitoring:
    temp = round(random.uniform(15, 45), 1)
    count += 1
    print(f"#{count} — الحرارة: {temp}°C")
    
    if temp > 40:
        print("🔴 تنبيه! إيقاف المراقبة.")
        monitoring = False
    
    if count >= 10:   # حد الأمان
        monitoring = False
⚠️ تحذير: دائماً ضع شرط إيقاف — وإلا سيعمل البرنامج إلى الأبد وستضطر لإيقافه يدوياً!

break و continue ⏹️
نتحكم في سلوك الحلقة بشكل أدق
# break — أوقف الحلقة فوراً
for i in range(100):
    temp = int(input("أدخل قراءة (-1 للخروج): "))
    if temp == -1:
        print("توقفت المراقبة.")
        break              # اخرج من الحلقة
    print(f"تم تسجيل: {temp}°C")

# continue — تجاهل هذه الدورة وانتقل للتالية
for i in range(10):
    if i % 2 == 0:
        continue           # تخطَّ الأعداد الزوجية
    print(f"فردي: {i}")  # يطبع 1,3,5,7,9

break

بناء الدوال
def — تعريف دالة 🔧
هيكل الدالة دائماً نفسه — اسم، مدخلات، كود، مخرج (return)
# دالة بسيطة
def greet(name):
    print(f"أهلاً يا {name}! 👋")

greet("زياد")     # → أهلاً يا زياد!
greet("سارة")     # → أهلاً يا سارة!

# دالة مع return — تُرجع قيمة
def check_sensor(value, min_val, max_val):
    if value > max_val:
        return "HIGH"
    elif value < min_val:
        return "LOW"
    else:
        return "OK"

# الاستخدام
status = check_sensor(38, 10, 35)
print(f"حالة الحرارة: {status}")  # → HIGH
💡 check_sensor() هذه هي نفس الدالة الموجودة في alerts.py في رصد (Rasd)!
