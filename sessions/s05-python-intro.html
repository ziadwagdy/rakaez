مقارنة الأرقام
عوامل المقارنة ⚖️
هذه الأدوات تسمح للبرنامج بمقارنة القيم — النتيجة دائماً True أو False
>
أكبر من
temp > 35
<
أصغر من
temp < 10
>=
أكبر أو يساوي
humid >= 80
<=
أصغر أو يساوي
soil <= 20
==
يساوي تماماً
status == "OK"
!=
لا يساوي
rain != 0
⚠️ انتبه: = للتعيين (حفظ قيمة) — == للمقارنة (هل يساوي؟)
اتخاذ القرار
if / elif / else 🔀
ثلاثة أجزاء لبناء أي قرار في Python
# درجة الحرارة — مثال رصد (Rasd) temp = float(input("أدخل درجة الحرارة: ")) if temp > 35: # الشرط الأول print("🔴 خطر! حار جداً") elif temp > 25: # شرط بديل print("🟡 دافئ — طبيعي") elif temp > 10: # شرط بديل آخر print("🟢 معتدل ومريح") else: # كل الحالات الأخرى print("🔵 بارد! تنبيه حرارة منخفضة")
📐 المسافة البادئة إلزامية — 4 مسافات بعد كل if/elif/else يجب أن يكون الكود داخلها منزاحاً للأمام
العوامل المنطقية
and / or / not 🔗
ندمج أكثر من شرط في قرار واحد — رصد (Rasd) يحتاجها دائماً
# and — الشرطان معاً يجب أن يكونا صحيحين if temp > 35 and humid > 80: print("🔴 حر ورطب — خطر مزدوج!") # or — يكفي شرط واحد صحيح if temp > 35 or temp < 10: print("⚠️ درجة حرارة خارج المعدل!") # not — يعكس النتيجة rain_ok = False if not rain_ok: print("🌧️ هناك مطر مكتشف!")

and

صحيح فقط إذا الاثنان صحيحان

or

صحيح إذا واحد أو أكثر صحيح

not

يحوّل True→False والعكس

حارس الغرفة الكامل 🛡️
نفس المنطق الموجود فعلاً في alerts.py داخل رصد (Rasd)!
# حارس الغرفة — كل الحساسات الأربعة temp = float(input("الحرارة (°C): ")) humid = float(input("الرطوبة (%): ")) water = int(input("حساس المطر (0-65535): ")) soil = int(input("رطوبة التربة (0-65535): ")) alerts = 0 if temp > 35 or temp < 10: print(f"🌡️ تنبيه حرارة: {temp}°C") alerts += 1 if humid > 80 or humid < 20: print(f"💧 تنبيه رطوبة: {humid}%") alerts += 1 if water < 20000: print("🌧️ تنبيه: مطر مكتشف!") alerts += 1 if soil < 20000: print("🌱 تنبيه: التربة جافة جداً!") alerts += 1 if alerts == 0: print("✅ كل شيء طبيعي — رصد (Rasd) مطمئن") else: print(f"⚠️ عدد التنبيهات: {alerts}")

if داخل if — حين تحتاج قراراً داخل قرار آخر
# تصنيف حالة الغرفة — درجتان من الخطورة if temp > 30: # دخلنا منطقة الحرارة العالية if humid > 70: print("🔴 خطر حريق محتمل! أغلق الحساسات!") else: print("🟠 حار لكن رطوبة معقولة — راقب الوضع") elif temp < 5: if humid > 60: print("❄️ بارد ورطب — خطر تكثف وصدأ!") else: print("🔵 بارد فقط — تنبيه عادي") else: print("✅ الغرفة في الحدود الآمنة")
