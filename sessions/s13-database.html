SQL — لغة البيانات
أوامر SQL الأساسية 🔍
ثلاثة أوامر تكفيك لاستعراض كل بيانات رصد (Rasd)
-- عرض كل القراءات

SELECT * FROM readings;


-- فلترة: الحرارة فوق 25 درجة فقط

SELECT * FROM readings

WHERE temperature > 25;


-- متوسط درجة الحرارة

SELECT AVG(temperature) AS avg_temp

FROM readings;


-- آخر 10 قراءات فقط

SELECT * FROM readings

ORDER BY timestamp DESC LIMIT 10;

Flask والـ Database
كيف يُخزّن Flask البيانات؟ 💾
في database.py — كل قراءة جديدة تُحفظ هكذا
# في database.py

import sqlite3


def save_reading(temp, hum, vib):

  conn = sqlite3.connect('room_guardian.db')

  cursor = conn.cursor()

  cursor.execute(

    "INSERT INTO readings (temperature, humidity, vibration)"

    " VALUES (?, ?, ?)",

    (temp, hum, vib)

  )

  conn.commit()

  conn.close()
استعلامات متقدمة
SQL أقوى مما تظن 🚀
بجملة واحدة تحصل على إجابة لأسئلة معقدة
-- أعلى درجة حرارة تم تسجيلها

SELECT MAX(temperature) FROM readings;


-- متوسط الرطوبة خلال آخر ساعة

SELECT AVG(humidity) FROM readings

WHERE timestamp > datetime('now', '-1 hour');


-- عدد مرات كشف الاهتزاز

SELECT COUNT(*) FROM readings

WHERE vibration = 1;
