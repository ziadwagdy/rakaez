# تعريف الدالة
def check_temperature(temp):
    if temp > 35:
        return "حرارة عالية ⚠️"
    elif temp < 10:
        return "برودة شديدة ❄️"
    else:
        return "طبيعي ✅"

# استخدام الدالة
result = check_temperature(28.5)
print(result)  # طبيعي ✅

result2 = check_temperature(38)
print(result2)  # حرارة عالية ⚠️

القوائم — مجموعة من القيم
# قائمة قراءات الحرارة
readings = [25.1, 27.3, 28.5, 26.0, 29.1]

# الوصول لعنصر
print(readings[0])   # 25.1 — الأول
print(readings[-1])  # 29.1 — الأخير

# طول القائمة
print(len(readings))  # 5

# أكبر وأصغر قيمة
print(max(readings))  # 29.1
print(min(readings))  # 25.1
