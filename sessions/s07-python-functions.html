def check_sensor(value, min_val, max_val):
    if value > max_val:
        return "HIGH"
    elif value < min_val:
        return "LOW"
    else:
        return "OK"

for i in range(3):
    result = check_sensor(38, 10, 35)
    print(f"القراءة {i+1}: {result}")

القوائم
List — تخزين أكثر من قيمة 📋
بدلاً من متغير واحد لكل قراءة — نجمعها كلها في قائمة واحدة
# قائمة بسيطة
readings = [32.1, 33.4, 31.8, 35.2, 29.9]

# الوصول بالرقم (يبدأ من 0)
print(readings[0])   # → 32.1 (الأول)
print(readings[-1])  # → 29.9 (الأخير)

# عمليات مفيدة
print(len(readings))    # → 5 (الطول)
print(max(readings))    # → 35.2 (الأعلى)
print(min(readings))    # → 29.9 (الأدنى)
print(sum(readings)/len(readings))  # المتوسط

# إضافة قراءة جديدة
readings.append(34.0)
print(f"عدد القراءات الآن: {len(readings)}")
🔢 الفهرسة تبدأ من 0 — القائمة [a, b, c]: a هو [0]، b هو [1]، c هو [2]
القواميس
Dict — بيانات بأسماء 📖
القاموس يربط مفتاحاً بقيمة — مثالي لتنظيم بيانات الحساسات!
# قاموس بيانات حساس واحد
reading = {
    "temperature": 34.2,
    "humidity":    65,
    "water_level": 45000,
    "soil_moisture": 38000,
    "location":    "الغرفة أ"
}

# الوصول بالمفتاح
print(reading["temperature"])  # → 34.2
print(reading["humidity"])     # → 65

# التعديل
reading["temperature"] = 35.1

# التكرار على المفاتيح
for key, value in reading.items():
    print(f"{key}: {value}")
💡 هذا القاموس هو بالضبط ما يرسله الـ Pico W لـ Flask في رصد (Rasd)!

import json

# من Python dict إلى نص JSON
data = {"temperature": 34.2, "humidity": 65}
json_text = json.dumps(data)
print(json_text)
# → '{"temperature": 34.2, "humidity": 65}'

# من نص JSON إلى Python dict
received = '{"temp": 36.5, "humid": 70}'
parsed = json.loads(received)
print(parsed["temp"])   # → 36.5
🌍 json.dumps() يحوّل dict لنص | json.loads() يحوّل نص لـ dict

import json

# ١. الـ Pico W يرسل هذا الـ JSON
incoming_json = '''
{
    "temperature": 34.2,
    "humidity": 75,
    "water_level": 15000,
    "soil_moisture": 42000
}
'''

# ٢. Flask يحوله لـ dict
data = json.loads(incoming_json)

# ٣. نفحص كل قيمة
if data["temperature"] > 35:
    print("🔴 تنبيه حرارة!")

if data["water_level"] < 20000:
    print("🌧️ تنبيه مطر!")

print(f"الحرارة: {data['temperature']}°C")
print(f"الرطوبة: {data['humidity']}%")
القوائم المتقدمة
قائمة من القواميس 📋📖
هكذا تخزن رصد (Rasd) قراءات متعددة في قاعدة البيانات
# قائمة من القراءات
history = [
    {"time": "10:00", "temp": 32.1, "humid": 60},
    {"time": "10:10", "temp": 33.5, "humid": 62},
    {"time": "10:20", "temp": 36.0, "humid": 78},
    {"time": "10:30", "temp": 34.2, "humid": 70},
]

# تحليل القراءات
for r in history:
    if r["temp"] > 35:
        print(f"⚠️ {r['time']}: حرارة عالية {r['temp']}°C")

# المتوسط
avg = sum(r["temp"] for r in history) / len(history)
print(f"متوسط الحرارة: {avg:.1f}°C")
🗄️ data.db في رصد (Rasd) يخزن بالضبط قائمة من هذه القواميس — لكن في SQLite!
