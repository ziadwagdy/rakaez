تحديث الرسم البياني
بيانات حية
let chart; // الرسم البياني

function updateChart() {
  fetch('/api/readings')
    .then(r => r.json())
    .then(data => {
      chart.data.labels   = data.map(r => r.timestamp.slice(11,16));
      chart.data.datasets[0].data = data.map(r => r.temperature);
      chart.update(); // أعِد الرسم
    });
}

setInterval(updateChart, 10000); // كل 10 ثوانٍ
