API Design
الـ endpoints الأربعة
# GET /api/latest  ← آخر قراءة
# GET /api/readings ← آخر 50 قراءة للرسم البياني
# GET /api/alerts   ← التنبيهات الفعّالة
# POST /api/data    ← Pico W يُرسل هنا

# مثال على GET /api/latest
@app.route('/api/latest')
def get_latest():
    reading = db.get_latest_reading()
    return jsonify({
        'temperature': reading['temperature'],
        'humidity': reading['humidity'],
        'water_level': reading['water_level'],
        'soil_moisture': reading['soil_moisture'],
        'timestamp': reading['timestamp']
    })
