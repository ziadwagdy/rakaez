JavaScript · البيانات الحية
جلب البيانات من الـ Backend ⚡
// في ملف app.js

function updateDashboard() {
  fetch('http://localhost:8080/api/latest')
    .then(response => response.json())
    .then(data => {
      // ضع القيم في البطاقات
      document.getElementById('temp').textContent  = data.temperature;
      document.getElementById('humid').textContent = data.humidity;
      document.getElementById('water').textContent = data.water_level;
      document.getElementById('soil').textContent  = data.soil_moisture;
    });
}

// شغّل الآن
updateDashboard();

// كرّر كل 3 ثوانٍ
setInterval(updateDashboard, 3000);
fetch() ترسل طلباً للـ API وتنتظر الرد · setInterval تعيد التشغيل كل X ميلي ثانية (3000 = 3 ثوانٍ)
JavaScript · منطق التنبيه
لون يتغير حسب القيمة 🔴🟢
function updateDashboard() {
  fetch('http://localhost:8080/api/latest')
    .then(response => response.json())
    .then(data => {
      const tempEl = document.getElementById('temp');
      tempEl.textContent = data.temperature;

      // غيّر اللون حسب القيمة
      if (data.temperature > 35) {
        tempEl.style.color = '#ff6b6b';   // أحمر — خطر
      } else if (data.temperature < 10) {
        tempEl.style.color = '#58a6ff';   // أزرق — بارد
      } else {
        tempEl.style.color = '#00e5c0';   // عادي
      }
    });
}
هذا نفس if/else الذي تعلمناه في Python — JS تستخدم نفس المنطق تماماً، فقط الكتابة مختلفة قليلاً
