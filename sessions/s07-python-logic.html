def check_alerts(temperature, humidity, water_level, soil):
    alerts = []
    
    if temperature > 35:
        alerts.append("⚠️ حرارة عالية: " + str(temperature) + "°C")
    if temperature < 10:
        alerts.append("❄️ برودة شديدة: " + str(temperature) + "°C")
    if humidity > 80:
        alerts.append("💧 رطوبة عالية جداً")
    if water_level < 20000:
        alerts.append("🌧️ مطر أو ماء مكتشف")
    
    return alerts

حلقة for — كرّر على كل عنصر
readings = [22.5, 36.1, 28.0, 38.5, 25.3]

for temp in readings:
    if temp > 35:
        print(f"تنبيه: {temp}°C — حرارة عالية!")
    else:
        print(f"طبيعي: {temp}°C")
